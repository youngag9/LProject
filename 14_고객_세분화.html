cluster_func <- function(all_client, type = c(1:10), num = c(1:1000) ) {
  
  # Class별 테이블 생성 
  Class_table <- all_client %>% 
    filter(CLNT_CL == type) 
  
  Class_table <- Class_table[9:18]
  
  set.seed(num)
  fit.kmeans <- kmeans(Class_table, centers = 2) 
  
  return(fit.kmeans)
}
