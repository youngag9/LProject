class PredModel(nn.Module):                        # nn.Module class를 상속
    def __init__(self, D_in,D_out):
        super(PredModel, self).__init__()           # 부모클래스( nn.Module) 의 __init__ 함수 호출
        self.linear1 = torch.nn.Linear(D_in,D_out)  # nn.linear 선형 연산 
        
    def forward(self,x):                            # forward 함수를 overriding 한다.
        out = self.linear1(x)                       # linear 함수를 통해 예측값을 계산
        return out
