2. Normal Client 고객 분류별 레이더 차트

## Normal Client의 고객분류간 상품 구매를 알 수 있는 레이더 차트 함수 생성
radar <- function(table, type1 = c(1:9), type2 = c(1:9) ) {
  t_radar <- table %>% 
    filter(CLNT_CL == type1 | CLNT_CL == type2) %>% 
    group_by(CLNT_CL) %>% 
    summarise(
      전자제품 = mean(C1_CT), 
      리빙 = mean(C2_CT),
      식품 = mean(C3_CT),
      스포츠 = mean(C4_CT),
      패션 = mean(C5_CT),
      뷰티 = mean(C6_CT), 
      유아용품 = mean(C7_CT), 
      취미용품 = mean(C8_CT),
      사무용품 = mean(C9_CT),
      상품권 = mean(C10_CT)
  )

  t_radar <- as.data.frame(t_radar)
  t_radar <- t_radar[-1] # CLNT_CL 지워줌 
  
  a <- range(t_radar)
  
  t_radar <- rbind(rep(a[2], 10), rep(a[1], 10), t_radar)
  
  rownames(t_radar) <- c('Max', 'Min', type1, type2)
  
  radarchart(t_radar, 
             axistype = 1,
             caxislabels = c( round(a[1], 2),  # 최소값
                              round((a[2]-a[1])*0.25, 2), # 25%
                              round((a[2]-a[1])*0.50, 2), # 50%
                              round((a[2]-a[1])*0.75, 2), # 75%
                              round(a[2], 2) ) # 100%
  )
  
  colors_in = c('black', 'red')
  
  # c <- ifelse(rownames(t_radar)[3] == '1', '10~20대 남성', '나머지')
  # print(c)
  
    a <- paste(ifelse(type1 == 1, '10~20대 남성',
                      ifelse(type1 == 2, '10대 여성',
                             ifelse(type1 == 3, '20대 여성',
                                    ifelse(type1 == 4, '30대 남성',
                                           ifelse(type1 == 5, '30대 여성',
                                                  ifelse(type1 == 6, '40~50대 남성',
                                                         ifelse(type1 == 7, '40~50대 여성',
                                                                ifelse(type1 == 8, '60~80대 남성', '60~80대 여성')))))))), '고객')
    
    b <- paste(ifelse(type2 == 1, '10~20대 남성',
                      ifelse(type2 == 2, '10대 여성',
                             ifelse(type2 == 3, '20대 여성',
                                    ifelse(type2 == 4, '30대 남성',
                                           ifelse(type2 == 5, '30대 여성',
                                                  ifelse(type2 == 6, '40~50대 남성',
                                                         ifelse(type2 == 7, '40~50대 여성',
                                                                ifelse(type2 == 8, '60~80대 남성', '60~80대 여성')))))))), '고객')
    
    
    legend(x='topright', y=1, 
           legend=c(a, b), 
           bty='n',pch=20, col=colors_in, text.col='black', cex=1,pt.cex=2 )
  


}

(1) 10~20대 남성 vs 10대 여성

# 10~20대 남/ 10대 여
radar(normal_client, 1, 2)

인사이트

10~20대 남성 고객과 10대 여성 고객 모두 패션의 매출이 가장 높았으며, 남성과 달리 여성의 경우 뷰티의 매출도 높은 것을 알 수 있다.

(3) 30대 남성 vs 30대 여성

  • 남녀 패션 매출은 비슷했으나 유아용품과 뷰티에서는 차이를 볼 수 있었다.

(5) 60~80대 남성 vs 60~80대 여성

인사이트

집단과 개인 선호 상품군 비교

# 자신의 클래스 번호, 클러스터링 번호, 고객 번호를 넣어서 고객 한 명의 성향을 확인해주는 
# radar라는 함수를 생성
radar <- function(table, class = c(1:10), CL_num = c(1:3), CLNT_Number = numeric ) {
  t_radar <- table %>% 
    filter(CLNT_CL == class & CL_num == Cluster_CL) %>% 
    summarise(
      전자제품 = mean(C1_CT), 
      리빙 = mean(C2_CT),
      식품 = mean(C3_CT),
      스포츠 = mean(C4_CT),
      패션 = mean(C5_CT),
      뷰티 = mean(C6_CT), 
      유아용품 = mean(C7_CT), 
      취미용품 = mean(C8_CT),
      사무용품 = mean(C9_CT),
      상품권 = mean(C10_CT)
    )
  
  person <- table %>% 
    filter(CLNT_ID == CLNT_Number) %>% 
    summarise(
      전자제품 = sum(C1_CT), 
      리빙 = sum(C2_CT),
      식품 = sum(C3_CT),
      스포츠 = sum(C4_CT),
      패션 = sum(C5_CT),
      뷰티 = sum(C6_CT), 
      유아용품 = sum(C7_CT), 
      취미용품 = sum(C8_CT),
      사무용품 = sum(C9_CT),
      상품권 = sum(C10_CT)
    )
  
  t_radar <- rbind(t_radar, person)

  a <- range(t_radar)
  
  t_radar <- rbind(rep(a[2], 10), rep(a[1], 10), t_radar)
  
  rownames(t_radar) <- c('Max', 'Min', 'Class_Average', 'CLNT_ID')
  
  x <- paste(
    (ifelse(class == 1, '10~20대 남성', 
            ifelse(class == 2, '10대 여성',
                   ifelse(class == 3, '20대 여성',
                          ifelse(class == 4, '30대 남성',
                                 ifelse(class == 5, '30대 여성',
                                        ifelse(class == 6, '40대 남성',
                                               ifelse(class == 7, '40대 여성',
                                                      ifelse(class == 8, '50~80대 남성', '50~80대 여성'))))))))
     ), 
     '+',
    (ifelse(CL_num == 1, '패션형',
            ifelse(CL_num == 2, '생활형', '뷰티형') )
      
    
      
    )
  )
  

  y <- paste('고객 ID', CLNT_Number)
  
  radarchart(t_radar, 
             axistype = 1,
             caxislabels = c( round(a[1], 2),  # 최소값
                              round((a[2]-a[1])*0.25, 2), # 25%
                              round((a[2]-a[1])*0.50, 2), # 50%
                              round((a[2]-a[1])*0.75, 2), # 75%
                              round(a[2], 2) ), # 100%
             title=paste(y,'와 ', x,'비교')
  )
  
  colors_in = c('black', 'red')

  legend(x='topright', y=1, 
         legend=c(x, y), 
         bty='n',pch=20, col=colors_in, text.col='black', cex=1, pt.cex=2 )
  
}
