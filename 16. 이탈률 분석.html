normal <- normal_client[c(1,29,30)]
# CLNT_ID, CLNT_CL, Cluster_num

session <- train_session
session$SESS_DT <- as.Date(session$SESS_DT, tz = '', tryFormats = c('%Y%m%d', '%Y-%m-%d'))

normalsess <- inner_join(normal, session, by='CLNT_ID')
# normal client들의 session 정보만 따로 담기

dfnor <- normalsess %>%  distinct(CLNT_ID, SESS_DT) %>%  arrange(CLNT_ID, SESS_DT)  
# 동일 일자에 생성한 세션 정보 제거

dfnor <- dfnor %>% group_by(CLNT_ID) %>% mutate(cnt = n())
# clnt_id 당 총 구매 일자 수 구하기

dfnor <-dfnor[order(dfnor$CLNT_ID, dfnor$SESS_DT),]
# clnt_id와 sess_dt로 정렬하기

dfnor$SESS_DTlag<-
  shift(dfnor$SESS_DT, n=1, fill=NA, type = "lag")
dfnor$SESS_DTdiff<-dfnor$SESS_DT - dfnor$SESS_DTlag
# SESS_DTlag 열에 앞 행에 있는 SESS_DT를 저장
# 앞 행에 있는 SESS_DT와의 간격을 계산

dfnor$SESS_DTfinal<-ifelse(dfnor$CLNT_ID == shift(dfnor$CLNT_ID, n=1, fill = NA, type = "lag"),
                        # TRUE: 앞 행의 CLNT_ID와 일치하면(동일 CLNT_ID의 구매일자가 맞으면)
                        dfnor$SESS_DT - shift(dfnor$SESS_DT, n=1, fill=NA, type = "lag"),
                        # FALSE: clnt_id가 한 행밖에 없을 때
                        NA)

a <- dfnor %>% filter(cnt >= 2)
a <- na.omit(a)
# 구매 일자가 2일 이상
# 즉, 평균 구매 주기를 구할 수 있는 경우
finalnor_df <- a %>% group_by(CLNT_ID) %>% mutate(mean_term =round(mean(SESS_DTfinal))) %>% select(CLNT_ID,cnt,mean_term) %>%  distinct(CLNT_ID,cnt,mean_term)
finalnor_df <- rbind(finalnor_df,b)
# a, b를 finalnor_df로 합쳐줌

최종 구매 경과일 구하기

dfnor<- dfnor %>% group_by(CLNT_ID) %>% mutate(last_day = max(SESS_DT))
lastnor <- dfnor %>% distinct(CLNT_ID, last_day)
# 마지막 구매 일자를 구함

finalnor_df <- inner_join(finalnor_df, lastnor, by='CLNT_ID')
finalnor_df$finalterm <- as.numeric(as.Date('2018-08-25') - finalnor_df$last_day)
# 마지막 구매 일자로 부터 2018년 8월 25일까지의 간격을 구함

finalnor_df$riskratio <- finalnor_df$finalterm/finalnor_df$mean_term
# riskratio=finalterm/mean_term

이탈 기준 정하기

aa <- finalnor_df %>% group_by(finalterm) %>% summarise(n=n())

# finalterm의 분포를 확인
ggplot(data=aa,aes(x=finalterm,y=n)) + 
  geom_line() + 
  ggtitle('최종 구매 경과일 당 고객 수') +
  xlab('최종 구매 경과일') +
  ylab('고객 수')+
  theme(plot.title = element_text(hjust=.5))

finalnor_df$churn <- ifelse(finalnor_df$finalterm>=53, 1, 0)

일반 고객과 VIP 고객 비교

비교를 위한 data frame 구성

diff <- finalnor_df
diff$class <-'일반고객'
diff <- rbind(diff, finalvip_df)
diff$class <- ifelse(is.na(diff$class)==T,'vip',diff$class)

최종 구매 경과일 비교

diff_finalterm<-diff %>% group_by(class, finalterm) %>% summarise(n=n())

ggplot(data=diff_finalterm, aes(x=finalterm, y=n, color=class)) + 
  geom_line() +
  ggtitle('최종 구매 경과일 당 고객 수 비교') +
  xlab('최종 구매 경과일') +
  ylab('고객 수')+
  theme(plot.title = element_text(hjust=.5))

인사이트

  • VIP 고객이 일반 고객보다 최종 구매 경과일이 증가할 수록 해당하는 고객 수가 급격하게 떨어진다.

  • 이는 VIP 고객의 최종 구매 경과일이 더 짧다는 것을 의미한다.

  • 따라서 VIP 고객의 충성도가 더 높다는 것을 알 수 있다.

이탈 가능성 비교

ggplot(data=diff, aes(x=riskratio, y=y, color=class)) + 
  geom_line() +
  ggtitle('위험비율에 대한 이탈 가능성 비교') +
  xlab('위험비율') +
  ylab('이탈 가능성') +
  theme(plot.title = element_text(hjust=.5))

인사이트

  • VIP고객들의 경우 약간만 오래 접속을 하지 않아도 이탈할 가능성이 높아진다.

  • 이탈 곡선이 완벽한 S자를 그리면서 오른쪽으로 치우칠수록 긍정적이라고 볼 수 있다.

  • 따라서 일반고객의 경우 VIP 고객보다 평균 접속 주기보다 더 오랜 기간 접속하지 않더라도 재접속하는 고객이 많다는 의미이다.

  • 위험비율이 0일 때, 즉 조사 당일 날 구매를 한 고객이더라도 일반 고객의 이탈 가능성이 VIP 고객보다 더 크다.
